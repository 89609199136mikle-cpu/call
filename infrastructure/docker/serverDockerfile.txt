# --- STAGE 1: Build ---
FROM node:20-alpine AS builder

# Устанавливаем рабочую директорию
WORKDIR /app

# Копируем файлы манифеста для кэширования слоев Docker
COPY package*.json ./

# Устанавливаем зависимости (только production для уменьшения размера)
RUN npm ci --only=production

# Копируем исходный код сервера
COPY server/ ./server/
COPY database/ ./database/

# --- STAGE 2: Production ---
FROM node:20-alpine

WORKDIR /app

# Устанавливаем ffmpeg — необходим для media-service (сжатие видео/аудио)
RUN apk add --no-cache ffmpeg

# Копируем node_modules и код из стадии сборки
COPY --from=builder /app/node_modules ./node_modules
COPY --from=builder /app/server ./server
COPY --from=builder /app/database ./database

# Railway автоматически назначает PORT, но мы задаем значение по умолчанию
ENV PORT=5000
ENV NODE_ENV=production

# Открываем порт для входящего трафика (API + WebSockets)
EXPOSE 5000

# Запускаем API Gateway — входную точку для всех запросов
CMD ["node", "server/gateway/apiGateway.js"]
