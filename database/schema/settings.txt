-- ==========================================
-- CRANEAPP DATABASE SCHEMA: SETTINGS
-- Описание: Пользовательские предпочтения и конфигурации
-- ==========================================

CREATE TABLE IF NOT EXISTS user_settings (
    user_id UUID PRIMARY KEY REFERENCES users(id) ON DELETE CASCADE,

    -- Внешний вид (Appearance.html)
    -- 'light', 'dark', 'system', 'telegram'
    theme_mode VARCHAR(20) DEFAULT 'system',
    accent_color VARCHAR(7) DEFAULT '#0088cc', -- HEX код Crane Blue
    font_size_ratio FLOAT DEFAULT 1.0, -- От 0.8 до 1.5
    wallpaper_url TEXT,

    -- Уведомления (Notifications.html)
    show_notifications BOOLEAN DEFAULT true,
    show_message_preview BOOLEAN DEFAULT true,
    notification_sound VARCHAR(50) DEFAULT 'default.mp3',
    is_vibration_enabled BOOLEAN DEFAULT true,

    -- Приватность и Безопасность (Privacy.html)
    -- 'everybody', 'contacts', 'nobody'
    privacy_last_seen VARCHAR(20) DEFAULT 'everybody',
    privacy_profile_photo VARCHAR(20) DEFAULT 'everybody',
    privacy_phone_number VARCHAR(20) DEFAULT 'contacts',
    is_two_factor_enabled BOOLEAN DEFAULT false,

    -- Данные и Хранилище (DataStorage.html)
    auto_download_media_mobile BOOLEAN DEFAULT false,
    auto_download_media_wifi BOOLEAN DEFAULT true,
    proxy_settings JSONB DEFAULT '{}',

    -- Язык (Language.html)
    language_code VARCHAR(5) DEFAULT 'en', -- 'en', 'ru', 'de' и т.д.

    -- Таймстамп обновления
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Индекс для аналитики (например, сколько пользователей используют темную тему)
CREATE INDEX IF NOT EXISTS idx_settings_theme ON user_settings(theme_mode);
