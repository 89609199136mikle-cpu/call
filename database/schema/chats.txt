-- Таблица чатов (Группы, Каналы, Приватные диалоги)
-- Путь: database/schema/chats.txt

CREATE TABLE IF NOT EXISTS chats (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- Тип сущности: 'private', 'group', 'channel'
    type VARCHAR(20) NOT NULL DEFAULT 'private',
    
    -- Метаданные (для групп и каналов)
    title VARCHAR(100),
    description TEXT,
    avatar_url TEXT,
    
    -- Ссылка на создателя (Admin)
    creator_id UUID REFERENCES users(id) ON DELETE SET NULL,
    
    -- Настройки приватности
    is_public BOOLEAN DEFAULT false, -- Для публичных каналов
    invite_link VARCHAR(50) UNIQUE,  -- Ссылка-приглашение
    
    -- Таймстампы
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Таблица участников чата (Связующая таблица Many-to-Many)
CREATE TABLE IF NOT EXISTS chat_participants (
    chat_id UUID REFERENCES chats(id) ON DELETE CASCADE,
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    
    -- Роль в конкретном чате: 'member', 'admin', 'owner'
    role VARCHAR(20) DEFAULT 'member',
    
    -- Настройки уведомлений для конкретного юзера в этом чате
    is_muted BOOLEAN DEFAULT false,
    
    -- Время входа (чтобы не видеть старые сообщения до вступления)
    joined_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    
    PRIMARY KEY (chat_id, user_id)
);

-- Индексы для реактивной скорости чатов
CREATE INDEX idx_participants_user_id ON chat_participants(user_id);
CREATE INDEX idx_chats_invite_link ON chats(invite_link) WHERE invite_link IS NOT NULL;
