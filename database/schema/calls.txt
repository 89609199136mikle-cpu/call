-- ==========================================
-- CRANEAPP DATABASE SCHEMA: CALLS
-- Описание: История аудио и видео вызовов
-- ==========================================

CREATE TABLE IF NOT EXISTS calls (
    -- Уникальный ID звонка
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- Кто инициировал вызов (ссылка на users.id)
    caller_id UUID NOT NULL,
    
    -- Кому поступил вызов (ссылка на users.id)
    receiver_id UUID NOT NULL,
    
    -- Тип вызова: 'audio' или 'video'
    call_type VARCHAR(10) NOT NULL DEFAULT 'audio',
    
    -- Статус вызова:
    -- 'initiated' - вызов начат (звонок идет)
    -- 'active'    - собеседник поднял трубку
    -- 'completed' - звонок успешно завершен
    -- 'missed'    - собеседник не ответил
    -- 'rejected'  - собеседник сбросил вызов
    -- 'busy'      - собеседник уже находится в другом звонке
    status VARCHAR(20) NOT NULL DEFAULT 'initiated',
    
    -- Временные метки для аналитики и UI
    started_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    connected_at TIMESTAMP WITH TIME ZONE, -- Когда статус стал 'active'
    ended_at TIMESTAMP WITH TIME ZONE,
    
    -- Длительность разговора в секундах
    -- Вычисляется как (ended_at - connected_at)
    duration INTEGER DEFAULT 0,
    
    -- Технические метаданные для WebRTC (Signaling)
    -- Храним версию протокола или регион сервера Railway
    metadata JSONB DEFAULT '{}',

    -- Внешние ключи с каскадным удалением при удалении пользователя
    CONSTRAINT fk_caller FOREIGN KEY (caller_id) REFERENCES users(id) ON DELETE CASCADE,
    CONSTRAINT fk_receiver FOREIGN KEY (receiver_id) REFERENCES users(id) ON DELETE CASCADE
);

-- ==========================================
-- ИНДЕКСЫ ДЛЯ ОПТИМИЗАЦИИ (Railway/PostgreSQL)
-- ==========================================

-- Быстрая выборка истории звонков для конкретного пользователя (экран calls.html)
CREATE INDEX IF NOT EXISTS idx_calls_user_history 
ON calls (caller_id, receiver_id, started_at DESC);

-- Быстрый поиск пропущенных вызовов для отображения счетчика в UI
CREATE INDEX IF NOT EXISTS idx_calls_missed_status 
ON calls (receiver_id) WHERE status = 'missed';
